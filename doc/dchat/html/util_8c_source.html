150 void
151 print_dchat_msg(char* nickname, char* msg, int out_fd)
152 {
153  dprintf(out_fd, "%s", ansi_clear_line());
154  dprintf(out_fd, "%s", ansi_cr());
155  dprintf(out_fd, "%s", ansi_color_bold_cyan()); // colorize msg
156  dprintf(out_fd, "%s> ", nickname); // print nickname
157  dprintf(out_fd, "%s", ansi_reset_attributes()); // reset color
158  dprintf(out_fd, "%s", msg); // print message
159 
160  // append \n if line was not terminated with \n
161  if (msg[strlen(msg) - 1] != '\n')
162  {
163  dprintf(out_fd, "\n");
164  }
165 
166  rl_forced_update_display(); // redraw stdin
167 }
